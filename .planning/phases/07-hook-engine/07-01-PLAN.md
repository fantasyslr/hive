       ```typescript
       export class HookEngine {
         private hooks: HookDefinition[] = [];
         private handlers = new Map<string, ActionHandler>();
         private boundListeners = new Map<string, (event: HiveEvent) => void>();

         constructor(
           private bus: EventBus,
           handlers: Record<string, ActionHandler>,
         ) {
           for (const [type, handler] of Object.entries(handlers)) {
             this.handlers.set(type, handler);
           }
         }

         loadConfig(raw: unknown): { ok: boolean; error?: string } — Zod-parse raw, on success store hooks and call resubscribe(). On failure return {ok: false, error: zodError.message}. Previous hooks remain active on failure (per D-04).

         private resubscribe(): void — bus.off all boundListeners, then for each unique `hook.on` value, bus.on with a handler that filters hooks for that event, evaluates conditions, dispatches matching actions.

         private async dispatch(hook: HookDefinition, event: HiveEvent): Promise<void> — look up handler by hook.action.type, call handler.execute(event, hook.action.params). Wrap in try/catch, log errors, never throw.

         registerHooks(): void — no-op (config loaded separately), just log "HookEngine ready". Exists to match service pattern.
       }
       ```
